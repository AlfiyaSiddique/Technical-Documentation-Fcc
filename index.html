  function alphabet(string){
      let newString = [];
      for (let i = 0; i < string.length; i++){
          let code = string.charCodeAt(i);
          let character = String.fromCharCode(code+1);

     newString = newString.join("");

  }

  • Expected Input/Output

  function matrix(matrixArr){
      if(matrixArr[0][1] == 0 && matrixArr[0][2] == 0 &&
      matrixArr[1][2] == 0 ){

          console.log("The given matrix is a lower triangular 
          matrix");
     }
     else{
         console.log("The given matrix is not a lower triangular
          matrix");
     }
  }

  matrix([[1,0,0],[4,22,0],[6,5,1]]);

  • Explanation

  • In the Matrix function, there is an argument called matrixArr which is a multi-dimensional array of 3 by 3 matrix element
  • Inside this function, In an if condition I checked whether the element of specific index's are zero or not. Those index's are (0,1) , (0,2) ,(1,2)
  • If they are zero then the matrix is a lower triangular matrix else it is not a lower triangular matrix.

  • Statement Explanation

    In this program you have to take a year as an argument and return the Statement whether the passed year is leap year or not. Know the conditions to determine whether a year is a leap year or not

  • Expected Input/Output

    Input: 2020

    2020 : Leap year

    Input: 2021

    2020 : Not a Leap year

  • Source Code

                                                                

                              
  function leapYear(year){
     if(year%4 == 0 || year%100 == 0 && 
     year%400 == 0){
        console.log(`${year} is a leap year`);
    }
    else{
        console.log(`${year} is not a leap year`);
    }
  }

  leapYear(2020);

  • Explanation

  • In the leapYear function, I have a parameter named year, which will be the number of year
  • Inside this function, in an if condition I have checked whether the year is evenly divisible by 4 OR it is evenly divisible by 100 as well as 400
  • If the year satisfies any of the two conditions then it is a leapYear
  • Else, it is not a leap year

          }

        for(let i = 0; i < link.length; i++){
                  if(link[i] == ''){
                      link.splice(i,1);

                  }
              }
